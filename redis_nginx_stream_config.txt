# Add this to the stream block in /etc/nginx/nginx.conf
# After the PostgreSQL configuration (before the closing brace of stream)

    # Redis TCP Passthrough
    upstream redis_backend {
        server 127.0.0.1:6380;
    }

    server {
        listen 6380;
        proxy_pass redis_backend;
        proxy_connect_timeout 10s;
        proxy_timeout 300s;
    }

# Commands to apply (run on server):
# sudo nano /etc/nginx/nginx.conf
# Add the above configuration inside the stream block
# sudo nginx -t
# sudo systemctl reload nginx

# After applying, the Redis URL will work as:
# redis://:Kesar302redis@arpansahu.space:6380
